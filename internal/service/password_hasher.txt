package service

import (
	"context"
	"errors"
	"runtime"

	"golang.org/x/crypto/bcrypt"
)

var ErrHasherQueueTimeout = errors.New("password hasher queue timeout")

type hashJob struct {
	ctx      context.Context
	password string
	resultCh chan hashResult
}

type hashResult struct {
	hash string
	err  error
}

type PasswordHasher struct {
	jobs chan hashJob
}

// NewPasswordHasher запускает фиксированный пул воркеров,
// разбирающих очередь bcrypt-задач.
func NewPasswordHasher(workers int) *PasswordHasher {
	if workers <= 0 {
		workers = runtime.NumCPU()
	}

	h := &PasswordHasher{
		// небольшой буфер, чтобы не блокировать вызывающего мгновенно
		// при кратковременных всплесках
		jobs: make(chan hashJob, workers*4),
	}

	for i := 0; i < workers; i++ {
		go h.worker()
	}

	return h
}

func (h *PasswordHasher) worker() {
	for job := range h.jobs {
		select {
		case <-job.ctx.Done():
			// клиент уже отвалился (таймаут/отмена) — не тратим CPU впустую
			job.resultCh <- hashResult{err: job.ctx.Err()}
			continue
		default:
		}

		hash, err := bcrypt.GenerateFromPassword([]byte(job.password), bcrypt.DefaultCost)
		job.resultCh <- hashResult{hash: string(hash), err: err}
	}
}

// Hash ставит пароль в очередь и блокируется, пока не освободится воркер.
func (h *PasswordHasher) Hash(ctx context.Context, password string) (string, error) {
	resultCh := make(chan hashResult, 1)

	select {
	case h.jobs <- hashJob{ctx: ctx, password: password, resultCh: resultCh}:
	case <-ctx.Done():
		return "", ErrHasherQueueTimeout
	}

	select {
	case res := <-resultCh:
		return res.hash, res.err
	case <-ctx.Done():
		return "", ErrHasherQueueTimeout
	}
}

// CompareHashAndPassword — bcrypt-сравнение тоже CPU-bound,
// под нагрузкой на /login это тот же риск.
func (h *PasswordHasher) Compare(ctx context.Context, hash, password string) error {
	type cmpResult struct{ err error }
	resultCh := make(chan cmpResult, 1)

	go func() {
		// само сравнение быстрее генерации, но всё равно CPU-bound,
		// так что тоже стоит гонять через тот же пул в высоконагруженных системах.
		// здесь для простоты — напрямую, можно завести отдельный jobs-канал по аналогии с Hash при необходимости.
		resultCh <- cmpResult{err: bcrypt.CompareHashAndPassword([]byte(hash), []byte(password))}
	}()

	select {
	case res := <-resultCh:
		return res.err
	case <-ctx.Done():
		return ctx.Err()
	}
}
